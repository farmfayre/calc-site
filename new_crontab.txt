# Farm Fayre weekly market pipeline crontab (v2.1)
# Install with: sudo -u farmfayre crontab /opt/farmfayre/scraper/crontab.txt
#
# Schedule: Mondays, staggered start ~7am IE, 2-hourly retries through ~3pm IE.
# weekly_run.sh self-staggers 0-59 min (stealth), so each tick fires at a
# random minute past the hour. Idempotent: once a week publishes successfully,
# later ticks that week are no-ops. The freshness guard blocks publishing until
# the source data is demonstrably settled (exit 4 -> the next tick retries).
# The final tick (14 UTC) sets FF_LAST_ATTEMPT=1 so that if the source still
# hasn't settled, you get an alert instead of silent retry.

PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin

# UTC times. Ireland summer (BST) = UTC+1, winter (GMT) = UTC.
# 6-14 UTC  ->  7am-3pm IE in summer, 6am-2pm IE in winter. Daytime either way.
0 6  * * 1 /opt/farmfayre/scraper/weekly_run.sh
0 8  * * 1 /opt/farmfayre/scraper/weekly_run.sh
0 10 * * 1 /opt/farmfayre/scraper/weekly_run.sh
0 12 * * 1 /opt/farmfayre/scraper/weekly_run.sh
0 14 * * 1 FF_LAST_ATTEMPT=1 /opt/farmfayre/scraper/weekly_run.sh

# Weekly log cleanup - first day of month at 3am UTC, keep last 12 weeks
0 3 1 * * find /opt/farmfayre/logs/weekly_*.log -mtime +84 -delete 2>/dev/null
